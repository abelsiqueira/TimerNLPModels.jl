Nonlinear optimization

using NLPModelsIpopt, ADNLPModels, TimerNLPModels
nlp = TimerNLPModel(ADNLPModel(x -> sum(x.^2), ones(3)))
stats = ipopt(nlp, print_level = 0)
"Execution stats: first-order stationary"
get_timer(nlp)
